\documentclass[tikz, margin=3.14mm]{standalone}

\usepackage{pgfplots}

\begin{document}
\begin{tikzpicture}
\begin{axis}[
    xmin=0, xmax=20,
    ymin=-1, ymax=1,
    xlabel={$x$},
    ylabel={$\frac{\cos(x)}{x}$},
    ]
    
\addplot [scatter,
    samples=100,
    domain=0:20,
    mark size=1pt,
    ] {cos(deg(x))/x};
    
\addplot [blue,
    samples=100,
    domain=0.1:20, 
    ] {1/x};

\addplot [blue,
    samples=100,
    domain=0.1:20, 
    ] {-1/x};
    
\end{axis}
\end{tikzpicture}
\end{document}
