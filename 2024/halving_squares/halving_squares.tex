\documentclass[margin=3.14mm]{standalone}
\usepackage{tikz}
\usetikzlibrary{backgrounds}

\begin{document}
\begin{tikzpicture}[scale=5, background rectangle/.style={fill=white}, show background rectangle]

    % draw square
    \draw (0, 0) rectangle (1, 1);

    % iterative procedure
    \foreach \level in {1, ..., 10} {
        \pgfmathsetmacro\x{1 - 0.5^(\level)} % specify x and y using the trick 
        \pgfmathsetmacro\y{1 - 0.5^(\level)} % 3/4, 7/8 etc. are partial sums 1/2 + 1/4 + ... so they're actually 1 - 1/2^n
        \pgfmathsetmacro\xmin{\x}
        \pgfmathsetmacro\ymin{1 - 0.5^(\level-1)} 
        \draw [densely dashed, gray] (\x, \ymin) -- (\x, 1);  % vertical
        \draw [densely dashed, gray] (\xmin, \y) -- (1, \y);  % horizontal
    }

\end{tikzpicture}
\end{document}
